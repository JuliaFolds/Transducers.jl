    Show method for transducers

    julia> Map(sin) |> Map(cos) |> Map(tan)
Map(sin) |>
    Map(cos) |>
    Map(tan)

julia> TeeZip(Map(sin) |> TeeZip(Map(tan)))
TeeZip(
    Map(sin) |>
        TeeZip(Map(tan))
)

julia> TeeZip(Map(sin) |> TeeZip(Map(tan) |> Filter(isfinite)) |> MapSplat(*))
TeeZip(
    Map(sin) |>
        TeeZip(
            Map(tan) |>
                Filter(isfinite)
        ) |>
        MapSplat(*)
)
